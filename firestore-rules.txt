// Firestore Security Rules for sinema-60677 project
// Copy these to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Movies collection - read for all, write for authenticated users only
    match /movies/{movieId} {
      allow read: if true; // Public read access
      allow write: if request.auth != null; // Requires authentication
    }
    
    // Admin collection (if needed)
    match /admin/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Default deny all other collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
