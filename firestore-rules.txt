// Firestore Security Rules for movie-catalog project
// Copy these to Firebase Console > Firestore Database > Rules
// TEMPORARY: Relaxed rules for NextAuth integration

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Movies collection - public read, open write for now
    match /movies/{movieId} {
      allow read: if true; // Public read access for catalog browsing
      allow write: if true; // Temporary: Allow all writes for NextAuth users
    }
    
    // Admin logs collection - allow all authenticated users
    match /admin_logs/{logId} {
      allow read, write: if true; // Temporary: Allow for debugging
    }
    
    // Rate limiting collection - allow all for now
    match /rate_limits/{userId} {
      allow read, write: if true; // Temporary: Allow for debugging
    }
    
    // Default allow for development
    match /{document=**} {
      allow read, write: if true; // Temporary: Allow all for development
    }
  }
}
