// Firestore Security Rules for movie-catalog project
// SECURE VERSION - Copy these to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if request comes from server (Firebase Admin SDK)
    function isServerRequest() {
      // Firebase Admin SDK requests don't have request.auth.uid
      // but have elevated privileges
      return request.auth == null || request.auth.uid == null;
    }
    
    // Movies collection - Public read, server-only write
    match /movies/{movieId} {
      // Anyone can read movies (for public catalog)
      allow read: if true;
      
      // Only server-side requests can write (Firebase Admin SDK from API routes)
      allow write: if isServerRequest();
    }
    
    // Admin logs collection - Server only
    match /admin_logs/{logId} {
      allow read, write: if isServerRequest();
    }
    
    // Rate limiting collection - Server only  
    match /rate_limits/{userId} {
      allow read, write: if isServerRequest();
    }
    
    // Default: Deny everything else
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
